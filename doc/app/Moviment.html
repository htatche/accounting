def fire
  unless self.ctimp.nil? or self.ctpref.nil? then
    if (self.ctimp * self.ctpref) < 0 then
      write_attribute(:deure_, self.ctimp)
    else
      write_attribute(:haver_, self.ctimp)
    end
  end
end
